\section{Conceptual Design}
\subsection{Requirements}
The requirements for the database are the following:
\begin{table}[H]
    \renewcommand{\arraystretch}{1.3}
    \begin{tabularx}{\textwidth}{|X|}
    \hline
    \multicolumn{1}{|c|}{\textbf{``SageFit'' -- Sports Equipment Booking System}} \\ \hline
    The company ``SageFit'' provides sports equipment setup and rental services for events and training sessions.
    Customers -- either individuals or companies -- can place bookings for sports equipment to be installed at a
    chosen event location. Each customer is identified by a unique code and has contact details including email,
    phone and address. Customers may own one or more event locations, each characterised by a full address
    (street, house number, postal code, city, province), an estimated setup time, and an equipment capacity.
    A booking records the date, type (one-time or recurring), duration, cost, booking method (phone, email,
    website, or postal mail), and contract type (one-time, seasonal, or promotional). Each booking is handled
    by exactly one installation team and is associated with one customer and one event location.
    Teams are identified by a unique code, a name, a maximum number of members, and a counter of
    installations performed. Each team is composed of members identified by a unique ID, name, surname, phone
    and email; every member belongs to exactly one team and the number of members in a team must not exceed
    its declared maximum. Teams operate from one or more depots, and a single depot can serve multiple teams
    (many-to-many relationship). Each depot is characterised by a name, city, province, address, region, and
    number of employees. Finally, each depot covers one or more municipalities (each municipality belongs to
    exactly one depot), identified by a name. \\ \hline
    \end{tabularx}
    \caption{Requirements}
\end{table}

\subsubsection{Level of Abstraction}
\begin{itemize}
    \item \textbf{Customer} -- entity
    \item \textbf{Event\_Location} -- entity, owned by a Customer
    \item \textbf{Booking} -- entity, connects Customer, Event\_Location, and Team
    \item \textbf{Team} -- entity
    \item \textbf{Member} -- entity, belongs to a Team
    \item \textbf{Depot} -- entity, N:M with Team
    \item \textbf{Municipality} -- entity, covered by a Depot
\end{itemize}

\subsubsection{Business Rules}
\begin{itemize}
    \item \textbf{BR1}: Customer.Type $\in$ \{`Individual', `Company'\}
    \item \textbf{BR2}: Booking.Type $\in$ \{`recurring', `one-time'\}
    \item \textbf{BR3}: Booking.Booking\_Method $\in$ \{`phone', `email', `website', `postal\_mail'\}
    \item \textbf{BR4}: Booking.Contract\_Type $\in$ \{`one-time', `seasonal', `promotional'\}
    \item \textbf{BR5}: Booking.Duration $> 0$; Booking.Cost $\geq 0$
    \item \textbf{BR6}: Team.Max\_Members $> 0$; Team.Installation\_Number $\geq 0$
    \item \textbf{BR7}: Depot.Number\_Of\_Employees $\geq 0$
    \item \textbf{BR8}: Event\_Location.Setup\_Time\_Estimate $> 0$; Event\_Location.Equipment\_Capacity $> 0$
    \item \textbf{BR9}: The number of members in a team must not exceed Team.Max\_Members
    \item \textbf{BR10}: Each booking is handled by exactly one team
    \item \textbf{BR11}: A booking cannot be placed at an event location with zero equipment capacity
    \item \textbf{BR12}: Team.Installation\_Number is automatically incremented each time a new booking is assigned to that team
\end{itemize}
